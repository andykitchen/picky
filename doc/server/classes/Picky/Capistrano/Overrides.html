def self.extended cap_config
  cap_config.instance_eval do

    namespace :deploy do
      tasks.delete :check
      tasks.delete :cold
      tasks.delete :migrations
      tasks.delete :migrate
      tasks.delete :upload

      namespace :web do
        tasks.delete :enable
        tasks.delete :disable
      end
    end

  end
end
