def dump_sqlite internal
  reset

  transaction do
    # Note: Internal structures need to
    #       implement each.
    #
    internal.each do |key, value|
      encoded_value = Yajl::Encoder.encode value
      db.execute 'insert into key_value values (?,?)', key.to_s, encoded_value
    end
  end
end
