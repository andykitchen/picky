def reload app_file = 'app'
  Dir.chdir PICKY_ROOT
  exclaim 'Reloading loader.'
  load_self
  exclaim 'Reloading framework.'
  load_framework
  exclaim "Reloading application in #{File.expand_path(app_file)}."
  load_application app_file
end
